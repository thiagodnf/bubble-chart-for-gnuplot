Bubble Chart for Gnuplot

set terminal pdf size 3.8,3.6 enhanced set output 'out.pdf' set xrange [{X_RANGE}] set yrange [{Y_RANGE}] set xtics ({X_TICS}) set ytics ({Y_TICS}) scale = {SCALE} set xtics rotate by {ROTATE} offset 0,0 circleColor = '{CIRCLE_COLOR}' textColor = '{TEXT_COLOR}' set grid unset key set multiplot layout 1,1 plot '-' using 1:2:($3*scale) with circles lt rgb circleColor fs solid border rgb circleColor fs transparent solid 0.5, \ '-' using 1:2:(sprintf("%d",$3)) with labels notitle textcolor rgb textColor {DATA_CIRCLE} EOF {DATA_TEXT} EOF
